# MegaMoE token sweep — run from inside the vLLM container shell
#
# IMPORTANT: multi-node sweeps call `srun` from inside the container.
# The vLLM image does not include Slurm — re-enter the container via the
# updated ./open_docker.sh (bind-mounts host srun + libs from the compute node).
#
#   exit          # leave container
#   ./open_docker.sh
#   which srun    # should print /usr/bin/srun (or similar)

cd /lustre/fsw/coreai_libraries_cudnn/mhoqueanik/flashinfer

IMG="/lustre/fsw/coreai_libraries_cudnn/mhoqueanik/docker_images/vllm-runtime-nsys-tmux.sqsh"
MOUNT="/lustre/fsw/coreai_libraries_cudnn/mhoqueanik:/lustre/fsw/coreai_libraries_cudnn/mhoqueanik"
WORKDIR="/lustre/fsw/coreai_libraries_cudnn/mhoqueanik"

export MASTER_ADDR="$(scontrol show hostnames "$SLURM_NODELIST" | head -n1)"
export MASTER_PORT=29500

# Shared container args for nested multi-node srun
SRUN_CONTAINER=(
  --srun-extra-args
  --container-image="$IMG"
  --container-mounts="$MOUNT"
  --container-workdir="$WORKDIR"
)

# --- 4 GPUs (1 node, no nested srun) ---
python benchmarks/moe_ep/bench_deepseek_v4_mega_moe_token_sweep.py \
  --nnodes 1 --nproc-per-node 4

# --- 8 GPUs (2 nodes x 4 GPUs) ---
python benchmarks/moe_ep/bench_deepseek_v4_mega_moe_token_sweep.py \
  --nnodes 2 --nproc-per-node 4 \
  "${SRUN_CONTAINER[@]}"

# --- 16 GPUs (4 nodes x 4 GPUs) ---
python benchmarks/moe_ep/bench_deepseek_v4_mega_moe_token_sweep.py \
  --nnodes 4 --nproc-per-node 4 \
  --token-sizes 2048 4096 8192 16384 \
  --warmup 10 --repeat 50 --verbose \
  "${SRUN_CONTAINER[@]}"

# --- dry-run: print nested srun/torchrun commands without executing ---
python benchmarks/moe_ep/bench_deepseek_v4_mega_moe_token_sweep.py \
  --dry-run --nnodes 4 --nproc-per-node 4 \
  "${SRUN_CONTAINER[@]}"
